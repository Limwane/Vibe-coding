<body>
  <div class="app">
    <h1>🏋️ Limwane 运动计划</h1>

    <div class="input-area">
      <input id="input" placeholder="输入运动，例如：跑步30分钟"
             onkeydown="if(event.key==='Enter') addTask()">
      <button onclick="addTask()">添加</button>
    </div>

    <ul id="list"></ul>

    <button class="clear" onclick="clearDone()">清除已完成</button>
  </div>

<script>
let tasks = JSON.parse(localStorage.getItem("limwane_tasks") || "[]");

/* 💾 保存数据 */
function save() {
  localStorage.setItem("limwane_tasks", JSON.stringify(tasks));
}

/* 🎨 渲染列表 */
function render() {
  const list = document.getElementById("list");
  list.innerHTML = "";

  tasks.forEach((t, index) => {
    const li = document.createElement("li");

    li.innerHTML = `
      <span class="${t.done ? 'done' : ''}">${t.text}</span>
      <div class="actions">
        <input type="checkbox" ${t.done ? "checked" : ""} onchange="toggle(${index})">
        <button class="delete" onclick="removeTask(${index})">删</button>
      </div>
    `;

    list.appendChild(li);
  });
}

/* ➕ 添加任务 */
function addTask() {
  const input = document.getElementById("input");
  const value = input.value.trim();

  if (!value) return;

  tasks.push({ text: value, done: false });
  input.value = "";

  save();
  render();
}

/* ✔️ 勾选状态 */
function toggle(i) {
  tasks[i].done = !tasks[i].done;
  save();
  render();
}

/* ❌ 删除任务 */
function removeTask(i) {
  tasks.splice(i, 1);
  save();
  render();
}

/* 🧹 清除已完成 */
function clearDone() {
  tasks = tasks.filter(t => !t.done);
  save();
  render();
}

/* 🚀 初始化渲染（关键修复） */
render();
</script>
